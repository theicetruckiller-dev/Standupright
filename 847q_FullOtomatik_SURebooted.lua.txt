local function sendWebhook(msg)
	if webhookURL == "" then return end
	local http = (syn and syn.request) or (http and http.request) or http_request or request
	if not http then return end
	local data = {
		content = "@everyone\n"..msg
	}
	http({
		Url = webhookURL,
		Method = "POST",
		Headers = {["Content-Type"] = "application/json"},
		Body = game:GetService("HttpService"):JSONEncode(data)
	})
end

-- Drag support
local dragging, dragInput, dragStart, startPos
Frame.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
		dragging = true
		dragStart = input.Position
		startPos = Frame.Position
		input.Changed:Connect(function()
			if input.UserInputState == Enum.UserInputState.End then
				dragging = false
			end
		end)
	end
end)
Frame.InputChanged:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
		dragInput = input
	end
end)
UIS.InputChanged:Connect(function(input)
	if input == dragInput and dragging then
		local delta = input.Position - dragStart
		Frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
	end
end)

-- Oto Farm Fonksiyonları (dummy örnek mantık, oyun içi eventlere özel kodlar ile genişletilmeli)
local function getMoneyAndLevel()
	local money = LocalPlayer.leaderstats and LocalPlayer.leaderstats:FindFirstChild("Money") and LocalPlayer.leaderstats.Money.Value or "?"
	local level = LocalPlayer.leaderstats and LocalPlayer.leaderstats:FindFirstChild("Level") and LocalPlayer.leaderstats.Level.Value or "?"
	return money, level
end

local function announceFound(typ, name)
	local money, level = getMoneyAndLevel()
	local msg = string.format("[FULL OTOMATİK]\nKategori: %s\nItem: %s\nOyuncu: %s\nSaat: %s\nPara: %s\nSeviye: %s", typ, name, LocalPlayer.Name, os.date("%X"), money, level)
	sendWebhook(msg)
end

local farmThread = nil
local function startAutoFarm()
	if farmThread then return end
	autoFarmActive = true
	statusText = "Başladı"
	StatusLab.Text = "Durum: "..statusText
	farmThread = coroutine.create(function()
		while autoFarmActive do
			-- 1. Boss kontrol
			for _,boss in ipairs(bosses) do
				if farmBosses[boss] and Workspace.Bosses:FindFirstChild(boss) then
					statusText = boss.." bulundu, saldırılıyor"
					StatusLab.Text = "Durum: "..statusText
					-- Oto saldırı (örnek, skill spam + m1)
					for skill,_ in pairs(selectedSkills) do
						ReplicatedStorage.Events:FindFirstChild(skill):FireServer()
						wait(0.15)
					end
					ReplicatedStorage.Events.M1:FireServer()
					wait(0.2)
					if webhookOptions.bossItem then
						for _,it in ipairs(bossItems) do
							if Workspace.ItemSpawns:FindFirstChild(it) then
								announceFound("Boss Item", it)
							end
						end
					end
				end
			end

			-- 2. Item kontrol
			for _,it in ipairs(worldItems) do
				if farmWorldItems[it] and Workspace.ItemSpawns:FindFirstChild(it) then
					statusText = it.." bulundu, alınıyor"
					StatusLab.Text = "Durum: "..statusText
					-- Oto pickup
					LocalPlayer.Character.HumanoidRootPart.CFrame = Workspace.ItemSpawns[it].CFrame
					wait(0.2)
					if webhookOptions.worldItem then
						announceFound("World Item", it)
					end
				end
			end

			-- 3. Dungeon farm (örnek, gerçek teleport ve farm eventleri eklenmeli)
			if selectedDungeon then
				statusText = "Dungeon: "..selectedDungeon.." farm"
				StatusLab.Text = "Durum: "..statusText
				-- Dummy: Dungeon teleport event
				if ReplicatedStorage:FindFirstChild("Events") and ReplicatedStorage.Events:FindFirstChild("TeleportDungeon") then
					ReplicatedStorage.Events.TeleportDungeon:FireServer(selectedDungeon)
				end
				-- Skill spam
				for skill,_ in pairs(selectedSkills) do
					ReplicatedStorage.Events:FindFirstChild(skill):FireServer()
					wait(0.15)
				end
				ReplicatedStorage.Events.M1:FireServer()
				wait(1)
			end
			wait(0.2)
		end
		statusText = "Durduruldu"
		StatusLab.Text = "Durum: "..statusText
	end)
	coroutine.resume(farmThread)
end

local function stopAutoFarm()
	autoFarmActive = false
	statusText = "Durduruldu"
	StatusLab.Text = "Durum: "..statusText
	farmThread = nil
end

StartBtn.MouseButton1Click:Connect(function()
	if autoFarmActive then
		stopAutoFarm()
		StartBtn.Text = "Başlat"
		StartBtn.BackgroundColor3 = Color3.fromRGB(0,255,127)
	else
		startAutoFarm()
		StartBtn.Text = "Durdur"
		StartBtn.BackgroundColor3 = Color3.fromRGB(255,85,127)
	end
end)
